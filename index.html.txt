                        let div = document.createElement('div');
                        div.className = 'search-item';
                        div.innerText = p.name;
                        div.onclick = function() {
                            resBox.classList.add('hidden');
                            document.getElementById('main-search').value = "";
                            openSearchOrder(p.name, catKey, prodKey);
                        };
                        resBox.appendChild(div);
                    }
                });
            });

            if(matchCount > 0) { resBox.classList.remove('hidden'); } 
            else { resBox.innerHTML = '<div style="padding:10px; color:#888; font-size:13px; text-align:center;">কোনো পণ্য পাওয়া যায়নি!</div>'; resBox.classList.remove('hidden'); }

            let thana = document.getElementById('loc-thana').value;

        function buildFullAddress() {
            let div = document.getElementById('loc-division').value;
            let dist = document.getElementById('loc-district').value;
            let thana = document.getElementById('loc-thana').value;
            let village = document.getElementById('loc-village').value;
            let moholla = document.getElementById('loc-moholla').value;
            
            let addressText = "";
            if(div) addressText += `বিভাগ: ${div}, `;
            if(dist) addressText += `জেলা: ${dist}, `;
            if(thana) addressText += `থানা: ${thana}, `;
            if(village) addressText += `গ্রাম: ${village}, `;
            if(moholla) addressText += `মহল্লা/রোড: ${moholla}`;
            
            document.getElementById('o-addr').value = addressText;
        }

        function submitOrd() {
            let qty = document.getElementById('o-qty').value;
            let total = document.getElementById('o-total').innerText;
            let addr = document.getElementById('o-addr').value;
            if(!addr) { alert("ঠিকানা প্রদান করুন"); return; }

            let orderData = {
                custName: user.name,
                custPhone: user.phone,
                product: selectedProduct.name,
                pack: selectedProduct.packs[selectPack].name,
                qty: qty,
                total: Number(total),
                address: addr,
                payMethod: document.getElementById('o-pay').value,
                note: document.getElementById('o-note').value,
                status: 'pending',
                time: new Date().toLocaleString()
            };

            database.ref('orders').push(orderData).then(() => {
                alert("আপনার অর্ডারটি সফলভাবে ডাটাবেজে সাবমিট হয়েছে!");
                backToMain();
            });
        }

        function loadAdminStats() {
            database.ref('orders').on('value', snapshot => {
                let data = snapshot.val() || {};
                let total = 0, done = 0, pending = 0;
                let listBox = document.getElementById('order-list-box');
                listBox.innerHTML = "";
                
                Object.keys(data).forEach(k => {
                    let o = data[k];
                    total += o.total;
                    if(o.status === 'done') done += o.total;
                    else pending += o.total;

                    listBox.innerHTML += `
                        <div class="order-card">
                            <p style="margin: 3px 0; font-size:11px; color:#c62828; font-weight:bold;">🕒 সময় ও তারিখ: ${o.time || 'N/A'}</p>
                            <p style="margin: 3px 0;"><b>ক্রেতা:</b> ${o.custName} (${o.custPhone})</p>
                            <p style="margin: 3px 0;"><b>অर्डर:</b> ${o.product} (${o.pack}) x ${o.qty}</p>
                            <p style="margin: 3px 0;"><b>ঠিকানা:</b> ${o.address} ${o.note ? '['+o.note+']' : ''}</p>
                            <p style="margin: 3px 0;"><b>মোট:</b> ৳${o.total} [${o.status}]</p>
                            <button class="btn-ok" onclick="updateStatus('${k}', 'done')">✔ ডেলিভারি</button>
                        </div>`;
                });
                document.getElementById('st-all').innerText = total;
                document.getElementById('st-ok').innerText = done;
                document.getElementById('st-pending').innerText = pending;
            });
        }

        function updateStatus(key, status) {
            database.ref('orders/' + key).update({ status: status }).then(() => alert("স্ট্যাটাস আপডেট হয়েছে"));
        }

        function renderDel() {
            let catKey = document.getElementById('del-sel-cat').value;
            let box = document.getElementById('del-list-box');
            box.innerHTML = "";
            if(!catKey || !productsData[catKey]) return;

            Object.keys(productsData[catKey]).forEach(key => {
                let p = productsData[catKey][key];
                box.innerHTML += `
                    <div class="list-item">
                        <span>${p.name}</span>
                        <button class="btn-del" onclick="deleteProduct('${catKey}', '${key}')">ডিলিট</button>
                    </div>`;
            });
        }

        function deleteProduct(catKey, key) {
            if(confirm("পণ্যটি ডিলিট করতে চান?")) {
                database.ref(`products/${catKey}/${key}`).remove().then(() => {
                    alert("ডিলিট সম্পন্ন হয়েছে");
                    renderDel();
                });
            }
        }
