        function refreshStats() {
            let total = 0, okTotal = 0, pendingTotal = 0;
            orders.forEach(o => {
                total += o.price;
                if(o.status === 'OK') okTotal += o.price;
                if(o.status === 'Pending') pendingTotal += o.price;
            });
            if(document.getElementById('st-all')) document.getElementById('st-all').innerText = total;
            if(document.getElementById('st-ok')) document.getElementById('st-ok').innerText = okTotal;
            if(document.getElementById('st-pending')) document.getElementById('st-pending').innerText = pendingTotal;
        }

        function getLocationGPS() {
            if (navigator.geolocation) {
                navigator.geolocation.getCurrentPosition(position => {
                    let lat = position.coords.latitude;
                    let lon = position.coords.longitude;
                    document.getElementById('o-addr').value += `📍 লাইভ লোকেশন লিংক: https://www.google.com/maps?q=${lat},${lon}\n`;
                    alert("জিপিএস লোকেশন সফলভাবে যুক্ত হয়েছে!");
                }, () => { alert("লোকেশন পারমিশন পাওয়া যায়নি!"); });
            } else { alert("আপনার ব্রাউজারে জিপিএস সাপোর্ট করে না!"); }
        }

        // গ্লোবাল সার্চ ফাংশন
        function liveSearchProducts() {
            let query = document.getElementById('main-search').value.toLowerCase().trim();
            let box = document.getElementById('search-results');
            box.innerHTML = "";
            if(!query) { box.classList.add('hidden'); return; }
            
            let found = false;
            Object.keys(db).forEach(cat => {
                db[cat].forEach((item, index) => {
                    if(item.name.toLowerCase().includes(query)) {
                        found = true;
                        box.innerHTML += `
                        <div class="search-item" onclick="selectSearchItem('${cat}', ${index})">
                            <span><b>${item.name}</b></span>
                            <span style="color:green; font-size:12px;">৳${item.p1}</span>
                        </div>`;
                    }
                });
            });
            box.classList.toggle('hidden', !found);
        }

        function selectSearchItem(cat, index) {
            document.getElementById('search-results').classList.add('hidden');
            document.getElementById('main-search').value = "";
            openOrder("সার্চ করা পণ্য", cat);
            document.getElementById('sel-prod').value = index;
            showProdDetail();
        }

        function submitOrd() {
            let qty = parseInt(document.getElementById('o-qty').value) || 1;
            let totalPrice = document.getElementById('o-total').innerText;
            let address = document.getElementById('o-addr').value.trim();
            let payment = document.getElementById('o-pay').value;
            let note = document.getElementById('o-note').value.trim();
            let packSize = selectedPackNum === 1 ? curItem.s1 : (selectedPackNum === 2 ? curItem.s2 : curItem.s3);

            if(!address) { alert("অনুগ্রহ করে ঠিকানা দিন!"); return; }

            document.getElementById('submit-btn').innerText = "...";
            let now = new Date();
            let timeStr = now.toLocaleDateString() + " | " + now.toLocaleTimeString();
            
            orders.push({ 
                uName: user.name, uPhone: user.phone,
                iName: `${curItem.name} (${packSize}) x ${qty}`, 
                price: parseInt(totalPrice), addr: address, pay: payment,
                note: note || 'নাই', time: timeStr, status: 'Pending' 
            });
            
            syncCloudData();

            // Formspree ইমেইল সাবমিট
            document.getElementById('fps-name').value = user.name;
            document.getElementById('fps-phone').value = user.phone;
            document.getElementById('fps-prod').value = `${curItem.name} (${packSize}) x ${qty}`;
            document.getElementById('fps-price').value = totalPrice + " ৳";
            document.getElementById('fps-addr').value = address;
            document.getElementById('fps-pay').value = payment;
            document.getElementById('fps-note').value = note || 'নাই';
            document.getElementById('fps-form').submit();

            alert("অর্ডারটি ইন্টারনেটে লাইভ সাবমিট হয়েছে!");
            document.getElementById('o-addr').value = "";
            document.getElementById('o-note').value = "";
            backToMain();
        }
